package com.kaiqkt.authentication.domain.services

import com.nimbusds.jose.JWSAlgorithm
import com.nimbusds.jose.jwk.KeyUse
import com.nimbusds.jose.jwk.RSAKey
import org.springframework.beans.factory.annotation.Value
import org.springframework.stereotype.Service
import java.security.KeyPair
import java.security.KeyPairGenerator
import java.security.PrivateKey

//utilizar hashicorp vault
@Service
class RsaKeyService(
    @param:Value("\${authentication.key-id}")
    private val keyId: String
) {
    //mover para um vault
    private val keyPair: KeyPair = KeyPairGenerator.getInstance("RSA").apply { initialize(2048) }.genKeyPair()

    fun publicJwk(): RSAKey {
        val jwt = RSAKey.Builder(keyPair.public as java.security.interfaces.RSAPublicKey)
            .privateKey(keyPair.private)
            .keyUse(KeyUse.SIGNATURE)
            .algorithm(JWSAlgorithm.RS256)
            .keyID(keyId)
            .build()

        return jwt.toPublicJWK() as RSAKey
    }

    fun privateKey(): PrivateKey = keyPair.private
}
